659  bool peekTrackingData(RemoteTrackingFrameInfo& infoOut, slamtec_aurora_sdk_errorcode_t* errcode = nullptr) {
660  slamtec_aurora_sdk_tracking_info_t trackingInfo;
661  slamtec_aurora_sdk_tracking_data_buffer_t bufferInfo;
662 
663 
664 
665  std::vector<uint8_t> imgbufferLeft, imgbufferRight;
666  std::vector< slamtec_aurora_sdk_keypoint_t> keypointBufferLeft, keypointBufferRight;
667 
668 
669  // fetch the image and kp count
670  memset(&bufferInfo, 0, sizeof(bufferInfo));
671  auto result = slamtec_aurora_sdk_dataprovider_peek_tracking_data(_sdk, &trackingInfo, &bufferInfo);
672 
673  if (result != SLAMTEC_AURORA_SDK_ERRORCODE_OK) {
674  if (errcode) {
675  *errcode = result;
676  }
677  return false;
678  }
679 
680 
681  // allocate buffer
682  imgbufferLeft.resize(trackingInfo.left_image_desc.data_size);
683  imgbufferRight.resize(trackingInfo.right_image_desc.data_size);
684  keypointBufferLeft.resize(trackingInfo.keypoints_left_count);
685  keypointBufferRight.resize(trackingInfo.keypoints_right_count);
686 
687  bufferInfo.imgdata_left = imgbufferLeft.data();
688  bufferInfo.imgdata_right = imgbufferRight.data();
689  bufferInfo.keypoints_left = keypointBufferLeft.data();
690  bufferInfo.keypoints_right = keypointBufferRight.data();
691 
692  bufferInfo.imgdata_left_size = imgbufferLeft.size();
693  bufferInfo.imgdata_right_size = imgbufferRight.size();
694  bufferInfo.keypoints_left_buffer_count = keypointBufferLeft.size();
695  bufferInfo.keypoints_right_buffer_count = keypointBufferRight.size();
696 
697 
698  result = slamtec_aurora_sdk_dataprovider_peek_tracking_data(_sdk, &trackingInfo, &bufferInfo);
699  if (errcode) {
700  *errcode = result;
701  }
702 
703  if (result == SLAMTEC_AURORA_SDK_ERRORCODE_OK) {
704  infoOut = std::move(RemoteTrackingFrameInfo(trackingInfo,
705  std::move(imgbufferLeft), std::move(imgbufferRight),
706  std::move(keypointBufferLeft), std::move(keypointBufferRight)));
707  }
708 
709  return result == SLAMTEC_AURORA_SDK_ERRORCODE_OK;
710  }
