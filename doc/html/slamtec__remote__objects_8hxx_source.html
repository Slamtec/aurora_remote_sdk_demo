81  bool fromLocatorString(const char* input) {
82  std::string protocol;
83  std::string ip;
84  uint16_t tport = SLAMTEC_AURORA_SDK_REMOTE_SERVER_DEFAULT_PORT;
85 
86  std::string inputWrapper = input;
87 
88  size_t protocol_pos = inputWrapper.find('/');
89  if (protocol_pos != std::string::npos) {
90  protocol = inputWrapper.substr(0, protocol_pos);
91  }
92  else {
93  protocol = SLAMTEC_AURORA_SDK_REMOTE_SERVER_DEFAULT_PROTOCOL;
94  protocol_pos = -1;
95  }
96 
97  size_t port_pos = inputWrapper.rfind(':');
98  if (port_pos != std::string::npos) {
99  ip = inputWrapper.substr(protocol_pos + 1, port_pos - protocol_pos - 1);
100  auto&& portStr = inputWrapper.substr(port_pos + 1);
101  tport = (uint16_t)std::stoi(portStr);
102  }
103  else {
104  ip = inputWrapper.substr(protocol_pos + 1);
105  }
106 
107  snprintf(protocol_type, sizeof(protocol_type), "%s", protocol.c_str());
108  snprintf(address, sizeof(address), "%s", ip.c_str());
109  this->port = tport;
110 
111  return true;
112  }
