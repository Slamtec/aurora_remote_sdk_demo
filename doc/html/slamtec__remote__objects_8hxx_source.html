96  bool fromLocatorString(const char* input) {
97  std::string protocol;
98  std::string ip;
99  uint16_t tport = SLAMTEC_AURORA_SDK_REMOTE_SERVER_DEFAULT_PORT;
100 
101  std::string inputWrapper = input;
102 
103  size_t protocol_pos = inputWrapper.find('/');
104  if (protocol_pos != std::string::npos) {
105  protocol = inputWrapper.substr(0, protocol_pos);
106  }
107  else {
108  protocol = SLAMTEC_AURORA_SDK_REMOTE_SERVER_DEFAULT_PROTOCOL;
109  protocol_pos = -1;
110  }
111 
112  size_t port_pos = inputWrapper.rfind(':');
113  if (port_pos != std::string::npos) {
114  ip = inputWrapper.substr(protocol_pos + 1, port_pos - protocol_pos - 1);
115  auto&& portStr = inputWrapper.substr(port_pos + 1);
116  tport = (uint16_t)std::stoi(portStr);
117  }
118  else {
119  ip = inputWrapper.substr(protocol_pos + 1);
120  }
121 
122  snprintf(protocol_type, sizeof(protocol_type), "%s", protocol.c_str());
123  snprintf(address, sizeof(address), "%s", ip.c_str());
124  this->port = tport;
125 
126  return true;
127  }

355  RemoteImageRef(const slamtec_aurora_sdk_image_desc_t& desc, const void * data)
356  : _desc(desc), _data(data)
357  {

363  const void* _data;

365  /*
366  * @brief The description of the image
367  */
368  const slamtec_aurora_sdk_image_desc_t & _desc;

370 
371 public:
379  template <typename T>
380  typename std::enable_if<std::is_same<T, cv::Mat>::value, bool>::type
381  toMat(T &mat)const {
382  if (isEmpty()) {
383  mat = T();
384  return false;
385  }
386 
387  switch (_desc.format)
388  {
389  case 0: // mono
390  mat = T(_desc.height, _desc.width, 0, (void*)_data, _desc.stride);
391  break;
392  case 1: // bgr
393  mat = T(_desc.height, _desc.width, 0x10, (void*)_data, _desc.stride);
394  break;
395  case 2: // rgba
396  mat = T(_desc.height, _desc.width, 0x18, (void*)_data, _desc.stride);
397  break;
398  case 3: // depth (float32)
399  mat = T(_desc.height, _desc.width, 5, (void*)_data, _desc.stride);
400  break;
401  case 4: // point3D (floatx3 XYZ)
402  mat = T(_desc.height, _desc.width, 0x15, (void*)_data, _desc.stride);
403  break;
404 
405 
406  default:
407  mat = T();
408  return false;
409  }
410 
411  return true;
412  }
413 
419  const std::array<float, 3>* toPoint3D() const {
420  if (_desc.format != 4) {
421  return nullptr;
422  }
423 
424  return reinterpret_cast<const std::array<float,3>*>(_data);
425  }
426 
427 
428 
429  size_t getPointCount() const {
430  return _desc.height * _desc.width;
431  }
432 
433 
434  bool isEmpty() const {
435  return (_data == nullptr || _desc.height == 0 || _desc.width == 0);
